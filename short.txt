SYNOPSIS
   ttybudget -a|--add AMOUNT [ADD_OPTIONS] 
   ttybudget -p|--print [PRINT_OPTIONS] 
   ttybudget -h|--help
   ttybudget -v|--version
   ttybudget CONFIG_OPTIONS [MAIN_OPTIONS]

OPTIONS
   MAIN OPTIONS
      -a AMOUNT, --add AMOUNT
      -p, --print
      -h, --help 
      -v, --version 

   ADD OPTIONS
      -t TAG_A,TAG_B, ..., --tags TAG_A,TAG_B, ...
      -m "MESSAGE", --message "MESSAGE"
      -d DATE, --date DATE

   PRINT OPTIONS
      -q TAG_A,TAG_B, ..., --query-tags TAG_A,TAG_B, ...
      -i BOUND1 [BOUND2], --interval BOUND1 [BOUND2]
      -r BOUND1 [BOUND2], --range BOUND1 [BOUND2]
      -s, --sort
      -n, --no-footer

   CONFIG OPTIONS
      --base-dir BASE_DIRECTORY
      --currency CURRENCY_CHARACTER
      --date-in-<iso,us>
      --date-out-<iso,us,long,abbr> 
      --use-file RECORDS_FILE
